ToDISC

@author: aarabil / Aarab Ilias

ESRB | Beyond Banking Team | Securitisation Monitor | EDW Data Production Pipeline 2021

Helper functions and classes to push deal submissions from a local environment to DISC:

This module implements various auxiliary functions/classes which are used to push the latest deal submissions from a local environment to DISC. Adds (minimal) pre-processing to the raw data when pushing to Pipeline 2A to ensure queryable tables in a DISC environment. Raw data is pushed through Pipeline 2B as a backup.

Including: * Module to connect to DISC environment, Sets the starting point for Pipelines 2A/2B * Pipelines 2A/2B * Pre-processing of raw data located 2A

Prerequisite: * User needs DISC access (IGAM request: AA DISC-PU-BDA)

class ToDISC:
    def to_disc(self):
        """Pushes the latest deal submissions from a local environment to DISC. Adds (minimal) pre-processing
        to the raw data when pushing to Pipeline 2A to ensure queryable tables in a DISC environment.
        Raw data is pushed through Pipeline 2B as a backup and will be available within the /share/EDW
        folder of the respective datalab.
        * User needs DISC access (IGAM request: AA DISC-PU-BDA)
        """
        ## Add decorator to disc.to_csv to ensure fallback when connection drops
        # -----------------------------------------------------------------------
        # whenever our external connection falls out when pushing to DISC, we'll wait
        # for a while and try again (up to 10 times). Each time we wait an additional
        # minute longer for the connection to restore again
        disc.to_csv = retry(
            wait=wait_incrementing(start=10, increment=10, max=100),
            stop=stop_after_attempt(10),
            retry=retry_if_exception_type(HTTPError)
            | retry_if_exception_type(ConnectionError),
        )(disc.to_csv)

        ## Define paths
        # --------------
        # Local
        path_root = self._path_root
        path_trackers = os.path.join(path_root, "trackers")
        path_local = os.path.join(path_root, "tmp_csv_files")
        # DISC
        path_disc_backup = self._path_disc_data.replace('lld', 'lld_backup')
        path_disc_main = self._path_disc_data
        if self._testrun: #verify testing dir
            if 'testing' not in path_disc_main:
                return print("**Something went wrong! No testing dir registered for testruns!**")

        ## Pipeline 2A & 2B
        # -------------------
        # Filenames of csv files that are currently stored in a local environment but haven't been processed yet
        filtr = (self._trackers.DISC_submissions.local == "TMP") & (
            self._trackers.DISC_submissions.processed == "NO"
        )
        CSVs_to_push = self._trackers.DISC_submissions.loc[
            filtr,
            "csvFileName",
        ]

        for eachFile in CSVs_to_push:
            # Files contain two headers: (i) Field Number and (ii) Variable Name
            # We import all data as object dtypes to process raw data with low latency
            # We use an encoding scheme that overlaps most EU languages, however the encoding
            # scheme messes up the name of the first variable in some cases (for some unknown reason) so we readjust the name
            csv_file = pd.read_csv(
                os.path.join(path_local, eachFile),
                header=[0, 1],
                dtype="object",
                encoding="ISO-8859-1",
            )
            csv_file.rename(columns={csv_file.columns[0][0]: "AR1"}, inplace=True)

            ## Pipeline 2B: Raw data stored as backup in DISC, gzipped
            # ------------------------------------------------------------
            deal_name = self._trackers.DISC_submissions.edcode[
                self._trackers.DISC_submissions.csvFileName == eachFile
            ].values[0]

            try:
                disc.to_csv(
                    df=csv_file,
                    path=os.path.join(path_disc_backup, deal_name, eachFile),
                    compression="gzip",
                    header=True,
                    index=False,
                )

            ## Generic exception catcher (in most cases ConnectionError when vpn/cnxn falls out)
            # --------------------------------------------------------------------------------------
            except:
                # Save tracker up until this point
                self._trackers.DISC_submissions.to_csv(
                    path_or_buf=os.path.join(path_trackers, "disc_submissions_rmb.csv"),
                    header=True,
                    index=False,
                )
                # Sleep for 10min and try again
                sleep(600)
                try:
                    disc.to_csv(
                        df=csv_file,
                        path=os.path.join(path_disc_backup, deal_name, eachFile),
                        compression="gzip",
                        header=True,
                        index=False,
                    )
                except:
                    # Save tracker up until this point and stop running for now
                    self._trackers.DISC_submissions.to_csv(
                        path_or_buf=os.path.join(
                            path_trackers, "disc_submissions_rmb.csv"
                        ),
                        header=True,
                        index=False,
                    )
                    print(">>> disc_submission tracker updated successfully!")
                    print("-------------------------------------------------")
                    return print(
                        "Uknown error has occured while running to_disc() to push new submissions to DISC. \n"
                        "Please verify a stable network connection, and run the pipelines again. \n"
                        "Pipelines will automatically continue from the last downloaded deal based on the saved trackers."
                    )

            # Update tracker accordingly
            self._trackers.DISC_submissions.loc[
                self._trackers.DISC_submissions.csvFileName == eachFile, "backup"
            ] = "YES"

            ## Pipeline 2A: Processed data stored in main DISC storage, gzipped
            # ------------------------------------------------------------------
            # Pre-processing
            csv_file_processed = self.preprocessing(csv_file)

            # Push to DISC environment
            try:
                disc.to_csv(
                    df=csv_file_processed,
                    path=os.path.join(path_disc_main, eachFile),
                    compression="gzip",
                    header=True,
                    index=False,
                )


            ## Generic exception catcher (in most cases ConnectionError when vpn/cnxn falls out)
            # --------------------------------------------------------------------------------------
            except:
                # Save tracker up until this point
                self._trackers.DISC_submissions.to_csv(
                    path_or_buf=os.path.join(path_trackers, "disc_submissions_rmb.csv"),
                    header=True,
                    index=False,
                )
                # Sleep for 10min and try again
                sleep(600)
                try:
                    disc.to_csv(
                        df=csv_file_processed,
                        path=os.path.join(path_disc_main, eachFile),
                        compression="gzip",
                        header=True,
                        index=False,
                    )
                except:
                    # Save tracker up until this point and stop running for now
                    self._trackers.DISC_submissions.to_csv(
                        path_or_buf=os.path.join(
                            path_trackers, "disc_submissions_rmb.csv"
                        ),
                        header=True,
                        index=False,
                    )
                    print(">>> disc_submission tracker updated successfully!")
                    print("-------------------------------------------------")
                    print(
                        "Uknown error has occured while running to_disc() to push new submissions to DISC. \n"
                        "Please verify a stable network connection, and run the script again. \n"
                        "The script will automatically continue from the last downloaded deal based on the saved tracker."
                    )

            # Update tracker accordingly
            self._trackers.DISC_submissions.loc[
                self._trackers.DISC_submissions.csvFileName == eachFile, "processed"
            ] = "YES"

            ## Delete csv_file from local environment when backup and processing to DISC are successful
            if (
                self._trackers.DISC_submissions.loc[
                    self._trackers.DISC_submissions.csvFileName == eachFile, "backup"
                ]
                == "YES"
            ).all() & (
                self._trackers.DISC_submissions.loc[
                    self._trackers.DISC_submissions.csvFileName == eachFile,
                    "processed",
                ]
                == "YES"
            ).all():

                os.remove(os.path.join(path_local, eachFile))
                self._trackers.DISC_submissions.loc[
                    self._trackers.DISC_submissions.csvFileName == eachFile, "local"
                ] = "deleted"

            print(">>>>", eachFile, "has been pushed through Pipelines 2A & 2B!")

        ## save tracker
        self._trackers.DISC_submissions.to_csv(
            path_or_buf=os.path.join(path_trackers, "disc_submissions_rmb.csv"),
            header=True,
            index=False,
        )
        print(">>> Uploading complete & disc_submission tracker updated successfully!")
