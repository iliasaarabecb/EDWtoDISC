CreateBackups

ESRB | Beyond Banking Team | Securitisation Monitor | EDW Data Production Pipeline 2021

Helper functions and classes to allow testruns to be run without conflicting with production:

Including: * Creation of a backup dir of /trackers; /tmp_csv_files and /dictionaries * Clears DISC submodule allowing testruns to be run * Moves back backup data to main DISC submodule of the edwapi

Prerequisite: * None

def mover(self, restore=False, backup_dir=None):
    """Helps to move/restore essential folders to/from a back-up directory. Used mainly when running tests to
    safely move existing folders to an isolated environment.

    Parameters
    ----------
    restore : bool, optional
        Whether to create a new backup storage or to restore a previous one.
        In case of True, backup_dir needs to be specified, by default False
    backup_dir : str, optional
        Path of backup directory to use to restore files.
        Must be used with restore set to True, by default None

    Returns
    -------
    str
        Path of newly created backup directory.
        Only returned when restore has been set to False.

    """
    ## Define paths
    # -------------
    path_root = self._path_root
    path_csv_files = os.path.join(path_root, "tmp_csv_files")
    path_trackers = os.path.join(path_root, "trackers")
    path_json = os.path.join(path_root, "dictionaries")
    back_up = r"backup_" + dt.now().replace(microsecond=0).strftime(
        "%Y_%m_%d__%H_%M_%S"
    )
    dir_backup_new = os.path.join(path_root, back_up)

    # for some reason os.makedirs() doesn't take in first path of list
    # so I add it twice in the list for now
    # TODO: fix issue
    paths = [
        path_csv_files,
        path_json,
        path_trackers,
        path_csv_files,
        dir_backup_new,
    ]
    ## Temporarily remove main folders within DISC submodule of edwapi
    # ---------------------------------------------------------------------
    if not restore:

        # Check whether directories exist, if not make new ones
        for everyPath in paths[:-1]:  # exclude backup dir
            try:
                os.makedirs(everyPath)
            except FileExistsError:
                pass  # directory already exists

        # Move main folders to backup directory
        for everyPath in paths[:-1]:  # exclude backup dir
            shutil.move(everyPath, paths[-1])
            os.makedirs(everyPath)  # create new empty ones

        self.move.dir_backup = back_up
        self.move.status = f"Mover has created a backup located at: \n {back_up}"
        self._trackers.DISC_submissions = None  # reset internal state
        print(f">>> Mover has created a backup located at: \n {back_up}")
        self.logs.loggers.backups.info(f"Backups created within {back_up}")

    ## Restore main directories within DISC submodule of edwapi form backup
    # ---------------------------------------------------------------------
    if restore:

        # Check whether backup_dir is inputted by usr
        if backup_dir == None:
            return print(
                "Please define folder name of back up storage within backup_dir arg!"
            )

        # Redefine paths
        backup_dir = os.path.join(self._path_root, backup_dir)
        path_csv_files = os.path.join(backup_dir, "tmp_csv_files")
        path_trackers = os.path.join(backup_dir, "trackers")
        path_json = os.path.join(backup_dir, "dictionaries")

        # Check whether directories exist in backup dir
        paths_backup = [path_csv_files, path_json, path_trackers]
        for everyPath in paths_backup:
            if os.path.exists(everyPath):
                pass
            else:
                return print(
                    "Following dir not found!",
                    everyPath,
                    "\n"
                    "Please only run with restore=True after creating a backup dir with restore=False.",
                )

        def copytree2(source, dest):
            os.mkdir(dest)
            dest_dir = os.path.join(dest, os.path.basename(source))
            shutil.copytree(source, dest_dir)

        # Move folders back to main dir of DISC submodule
        for everyPath in paths[:-1]:
            if os.path.exists(everyPath):
                shutil.rmtree(everyPath)
                os.makedirs(everyPath)

        self.move.dir_backup = None
        self.move.status = (
            f"Mover has restored previously backup folders from {backup_dir}"
        )
        self.move.restore_meta = dir_util.copy_tree(
            backup_dir, path_root, verbose=0
        )

        print(f">>> successfully restored backups from {backup_dir}")
        self.logs.loggers.backups.info(f"Backups restored from within {backup_dir}")
