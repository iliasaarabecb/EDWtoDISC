Helper functions and classes to push deal submissions from a DISC environment to an Impala/Parquet table:

This module implements various auxiliary functions/classes which are used to push the latest deal submissions from a DISC environment to an Impala and/or Parquet table. Sets the stage for Pipelines 3A and 4A.

Including: * Module to connect to DISC * Impala compatible queries to create a TEXTFILE TABLE * Impala compatible queries to create a PARQUET TABLE

Prerequisite: * User needs DISC access (IGAM request: AA DISC-PU-BDA) * DISC environment storing the data (Pipeline 2A) needs to grant writing access to Impala * JSON file with a dictionary mapping Field Numbers to Variable Names * JSON file with a dictionary mapping Variables to Pandas compatible D-types * JSON file with a dictionary mapping Variables to Impala+Parquet compatible D-types

def create_tables(
    self,
    new_tables=False,
):
    """Pushes the latest deal submissions from a DISC environment to an Impala
    and/or Parquet table. Sets the stage for Pipelines 3A and 4A.
    * User needs DISC access (IGAM request: AA DISC-PU-BDA)
    * DISC environment storing the data (Pipeline 2A) needs to grant writing access to Impala
    * JSON file with a dictionary mapping Field Numbers to Variable Names
    * JSON file with a dictionary mapping Variables to Pandas compatible D-types
    * JSON file with a dictionary mapping Variables to Impala+Parquet compatible D-types
    JSON files can be found back in /dictionaries subfolder of the DISC submodule of edwapi. In
    case subfolder is empty, please run create_dictionaries() first from the create_dictionaries
    submodule to get fresh dictionaries.


    Parameters
    ----------
    new_tables : bool, optional
        Whether to create new tables (True) or to update existing tables (False), by default False

    """
    ## Add decorator to disc._cursor.execute to ensure fallback when connection drops
    # -----------------------------------------------------------------------
    # whenever our external connection falls while trying to connect to DISC/Impala, we'll wait
    # for a while and try again (up to 10 times). Each time we wait an additional
    # minute longer for the connection to restore again
    # disc._cursor.execute= retry( wait= wait_incrementing(start=10, increment=10, max=100) ,stop=stop_after_attempt(10),
    #                   retry= retry_if_exception_type(HTTPError) | retry_if_exception_type(ConnectionError) )(disc._cursor.execute)

    ## Define path and staging table name
    # -------------
    from connectors import disc

    success = False
    path_root = self._path_root
    path_json = os.path.join(path_root, "dictionaries")
    path_trackers = os.path.join(path_root, "trackers")
    tbl_base_name = self._tbl_name + "_dp_l01_landing"

    ## Initizalize new base (TEXTFILE) and parquet (PARQUET) table
    # ------------------------------------------------------------
    if new_tables:


        # Make sure that user has writing access to the underlying disc data path
        verification= input(f"""Please make sure to grant Impala writing rights to the disc data repo:\n
        {self._path_disc_data} \n
        See the troubleshooting page for more information. 
        Return [yes] after enabling writing rights. [y/n] """)

        if verification.lower() in ['n', 'no']: 
            return print(f"""\n Please make sure to grant Impala writing rights to the disc data repo:\n
                    {self._path_disc_data} \n
                    and run Pipelines 3A and 4A again.""")


        # Get dictionary of Impala+Parquet compatible data types
        with open(os.path.join(path_json, "rmbs_lln_dtypes_impala.json"), "r") as f:
            data_types_impala = json.load(f)

        # Create one large string matching all variables with the right data type
        order = tuple(data_types_impala.keys())
        data_types_impala_strings = ", ".join(
            "{} {}".format(k, data_types_impala[k])
            for k in sorted(data_types_impala, key=order.index)
        )

        # Base table initializer
        tbl_base_initialize = [
            """DROP TABLE IF EXISTS {0}.{1}""".format(self._datalab, tbl_base_name),
            """CREATE {3} TABLE {0}.{1} ({2})
        ROW FORMAT DELIMITED
        FIELDS TERMINATED BY ','
        STORED AS TEXTFILE TBLPROPERTIES ("skip.header.line.count" = "1")
        """.format(
                self._datalab,
                tbl_base_name,
                data_types_impala_strings,
                self._tbl_type,
            ),
            """LOAD DATA INPATH '{2}' INTO TABLE {0}.{1}""".format(
                self._datalab, tbl_base_name, self._path_disc_data
            ),
            """ COMPUTE STATS  {0}.{1}""".format(self._datalab, tbl_base_name),
            """REFRESH {0}.{1}""".format(self._datalab, tbl_base_name),
        ]

        # Parquet table initializer
        tbl_parquet_initialize = [
            """DROP TABLE IF EXISTS {0}.{1}""".format(
                self._datalab, self._tbl_name
            ),
            """CREATE TABLE {0}.{1} STORED AS PARQUET 
        TBLPROPERTIES ("parquet.compress"= "SNAPPY") AS 
        SELECT * FROM {0}.{2}""".format(
                self._datalab, self._tbl_name, tbl_base_name
            ),
            """COMPUTE STATS {0}.{1}""".format(self._datalab, self._tbl_name),
            """REFRESH {0}.{1}""".format(self._datalab, self._tbl_name),
        ]

        # Connect to Impala and execute queries
        try:
            print(">>> Utilizing Impala to execute queries...")
            for eachQuery in tbl_base_initialize + tbl_parquet_initialize:
                print(f">>> Executing through Impala: {eachQuery}")
                disc._cursor.execute(eachQuery)
            del disc  # close connection to DISC environment
            print(">>> Pipeline 3A and 4A successfully finished!")
            success = True
        except pyodbc.Error as e:
            print(f"""Please make sure you have WRITING rights to the underlying disc data folder: {self._path_disc_data} \n 
            See the troubleshooting section for more information \n 
            Export Impala error: \n 
            {e}""")
        except:
            # Sleep for 10min and try again
            sleep(600)
            try:
                print(">>> Utilizing Impala to execute queries...")
                for eachQuery in tbl_base_initialize + tbl_parquet_initialize:
                    print(f">>> Executing through Impala: {eachQuery}")
                    disc._cursor.execute(eachQuery)
                del disc  # close connection to DISC environment
                print(">>> Pipeline 3A and 4A successfully finished!")
                success = True
            except:
                # Stop running for now
                print(
                    "Uknown error has occured while running create_tables(). \n"
                    "Please verify a stable network connection, and run the function directly again (instead of running entire script again). \n"
                )

        # Update disc_submission tracker
        if success:
            self._trackers.DISC_submissions.loc[
                (self._trackers.DISC_submissions.backup == "YES")
                & (self._trackers.DISC_submissions.processed == "YES")
                & (self._trackers.DISC_submissions.local == "deleted"),
                ["DISC", "parquet"],
            ] = ["YES", "YES"]
            self._trackers.DISC_submissions.to_csv(
                path_or_buf=os.path.join(path_trackers, "disc_submissions_rmb.csv"),
                header=True,
                index=False,
            )
            print(">>> disc_submission tracker has been updated!")
            if self._testrun:
                self.logs.loggers.pipelines.info(
                    "Successfully ran pipelines 3A & 4A during testrun with new tables"
                )
    ## Update existing base (TEXTFILE) and parquet (PARQUET) table with new data
    # ---------------------------------------------------------------------------
    if not new_tables:
        # Update base table
        tbl_base_update = [
            """LOAD DATA INPATH '{2}' 
        INTO TABLE {0}.{1}""".format(
                self._datalab, tbl_base_name, self._path_disc_data
            ),
            """COMPUTE STATS {0}.{1}""".format(self._datalab, tbl_base_name),
            """REFRESH {0}.{1}""".format(self._datalab, tbl_base_name),
        ]

        # Update parquet table and refreeze
        tbl_parquet_update = [
            """DROP TABLE IF EXISTS  {0}.{1}""".format(
                self._datalab, self._tbl_name
            ),
            """CREATE TABLE {0}.{1} STORED AS PARQUET 
        TBLPROPERTIES ("parquet.compress"= "SNAPPY") AS 
        SELECT * FROM {0}.{2}""".format(
                self._datalab, self._tbl_name, tbl_base_name
            ),
            """COMPUTE STATS {0}.{1}""".format(self._datalab, self._tbl_name),
            """REFRESH {0}.{1}""".format(self._datalab, self._tbl_name),
        ]

        try:
            print(">>> Utilizing Impala to execute queries...")
            for eachQuery in tbl_base_update + tbl_parquet_update:
                disc._cursor.execute(eachQuery)
                print(f">>> Executing through Impala: {eachQuery}")
            del disc  # close connection to DISC environment
            print(">>> Pipeline 3A and 4A successfully finished!")
            success = True
        except:
            # Sleep for 10min and try again
            sleep(600)
            try:
                print(">>> Utilizing Impala to execute queries...")
                for eachQuery in tbl_base_update + tbl_parquet_update:
                    disc._cursor.execute(eachQuery)
                    print(f">>> Executing through Impala: {eachQuery}")
                del disc  # close connection to DISC environment
                print(">>> Pipeline 3A and 4A successfully finished!")
                success = True
            except Exception as e:
                if "no visible files" in str(e):
                    print("---------------------------------------------------")
                    print(
                        f"No data was retrieved in {self._path_disc_data}. Please re-run pipelines 1A and 2A to populate the remote Hadoop cluster!"
                    )
                else:  # most likely unstable or overcrowded connection from DISC
                    # Stop running for now
                    print("-------------------------------------------------")
                    print(
                        "Uknown error has occured while running create_tables(). \n"
                        "Please verify a stable network connection, and run pipelines 3A and 4A again (no need for running previous pipelines again). \n"
                    )

        # Update disc_submission tracker if pushed successfully
        if success:
            self._trackers.DISC_submissions.loc[
                (self._trackers.DISC_submissions.backup == "YES")
                & (self._trackers.DISC_submissions.processed == "YES")
                & (self._trackers.DISC_submissions.local == "deleted"),
                ["DISC", "parquet"],
            ] = ["YES", "YES"]
            self._trackers.DISC_submissions.to_csv(
                path_or_buf=os.path.join(path_trackers, "disc_submissions_rmb.csv"),
                header=True,
                index=False,
            )
            print(">>> disc_submission tracker has been updated!")
            if self._testrun:
                self.logs.loggers.pipelines.info(
                    "Successfully ran pipelines 3A & 4A during testrun with updated tables"
                )
