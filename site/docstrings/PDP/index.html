Main class to initialize and maintain Pipelines in production: See README.md for PDP information

The Pipelines class is used to initialize and maintain the EDW Data Production Pipeline 2021 which pushes deal submissions stored in EDW's database to a Parquet optimized table within DISC, leveraging HDFS and queryable through Impala.

Includes the full set of pipelines; 1->2A (,2B) -> 3A -> 4A.

Keeps track of the latest deal submissions from EDW's database and updates the Parquet table in DISC accordingly.

Production is static for the moment due to security concerns and the adhoc nature of the Pipelines. The accompanying main.py should be run monthly to keep Parquet table well updated.

Including: * Pipeline 1 * Pipeline 2A + 2B * Pipeline 3A + 4A

def extract_edw_deals(self):
    """Extracts the names of the unique set of deals available at EDW"""
    edw_submissions = self._trackers.EDW_submissions
    self.unique_edw_deals = edw_submissions.edcode.unique()
    self.edw_deals = [
        edw_submissions[edw_submissions.edcode == everyDeal]
        for everyDeal in self.unique_edw_deals
    ]

def from_DARWIN(
    cls,
    datalab="lab_org_esrb",
    tbl_name="edw_rmbs_loan",
    abs_type= "rmbs", #currently only rmbs, 
    data_level= "lld", #currently only lld
    tbl_type="EXTERNAL",
    # DARWIN specific input
    file_name="disc_submissions_rmb.csv",
    darwin_root_folder_id=290298255,
    darwin_sub_folder="EDWtoDISC",
):
    """Alternative constructor that initializes a Pipelines object with an existing submission tracker located in DARWIN.

    Returns
    -------
    Pipelines object
        Pipelines based on an existing submission tracker
    """

    # Initiate alternative constructor
    obj_from_darwin = cls(datalab, tbl_name, abs_type, data_level, tbl_type)

    # Delete current tracker if available
    try:
        os.remove(
            os.path.join(obj_from_darwin._path_trackers, "disc_submissions_rmb.csv")
        )
    except OSError:
        pass
    # Push tracker from DARWIN into local env
    obj_from_darwin.download_DARWIN(
        local_path=obj_from_darwin._path_trackers,
        file_name=file_name,
        darwin_root_folder_id=darwin_root_folder_id,
        darwin_sub_folder=darwin_sub_folder,
    )

    # Load in tracker
    obj_from_darwin._trackers.DISC_submissions = pd.read_csv(
        os.path.join(obj_from_darwin._path_trackers, "disc_submissions_rmb.csv"),
        header="infer",
    )

    """Set of user input requests to initialize Pipelines accordingly.

    # main user args
    yes = ["y", "yes"]
    no = ["n", "no"]

    # Testrun request
    TESTRUN = input("Run a testrun? [y/n] ").lower()
    while TESTRUN not in yes + no:
        TESTRUN = input("Run a testrun? [y/n] ").lower()
    if TESTRUN in yes:
        print(">>> Pipelines will be created within a testrun environment.")
    elif TESTRUN in no:
        print(">>> Pipelines will be executed in production mode.")

    if TESTRUN in no:
        # New tables request
        NEW_TABLES = input(
            "Create new tables? If no, existing tables will be "
            "updated with latest deal submissions. [y/n] "
        ).lower()
        while NEW_TABLES not in yes + no:
            NEW_TABLES = input(
                "Create new tables? If no, existing tables will be "
                "updated with latest deal submissions. [y/n] "
            ).lower()
        if NEW_TABLES in ["y", "yes"] and TESTRUN in ["n", "no"]:
            verify = input(
                "Are you sure you want to delete existing DISC tables "
                "and create new ones (not recommended!)? [y/n] "
            ).lower()
            if verify in ["y", "yes"]:
                print(
                    ">>> Existing tables will be deleted and replaced with new ones."
                )
            else:
                NEW_TABLES = "no"
                print(
                    ">>> Existing tables wil be updated with latest deal submissions."
                )
        else:
            print(
                ">>> Existing tables wil be updated with latest deal submissions."
            )
    else:
        NEW_TABLES = None

    # TODO: remov hardcoded type and level
    PATHS = input("Use ESRB datalab? [y/n]: ").lower()
    while PATHS not in yes + no:
        PATHS = input("Use standard ESRB datalab paths? [y/n]: ").lower()
    if PATHS in yes:
        datalab = "lab_org_esrb"
        tbl_name = "edw_rmbs_loan"
        abs_type = 'rmbs'
        data_level = 'lld'
        path_datalab = f"/data/lab/{datalab[4:]}" #path_datalab

        datalab = input("Which datalab would you like to use? ")

        abs_type = 'rmbs'
        data_level = 'lld'
        path_datalab = f"/data/lab/{datalab[4:]}" #path_datalab
        path_disc_data = f"{path_datalab}/share/EDW/testing/{abs_type}/{data_level}"
        tbl_type = "EXTERNAL"

    return ToNameSpace(

                "datalab": datalab,

def run_pipeline_1A(self, deal):
    """Activates pipeline 1A, that runs between EDWs database and the user's local environment

    Parameters
    ----------
    deal : str
        The deal of which the submissions will be downloaded.
    """

    self.title_generator(f"Deal: {deal.edcode.iloc[0]}")

    ## Download new deals
    # --------------------
    print(f"\n >> Initializing Pipeline 1A")
    print(f">>> Accessing available submissions...")

    # Constrain number of submissions during testrun
    if self._testrun and len(deal) > 2:
        deal = deal[0:2]

    # Ensure proxy_server is running
    if not self._connections.proxies.HTTPS:
        self.initialize_proxy()

    # Check whether the disc_sumbission tracker has already been inititialized
    if isinstance(self._trackers.DISC_submissions, pd.DataFrame):

        # If so, get current list of deals within pipeline
        unique_disc_deals = self._trackers.DISC_submissions.edcode.unique()

        # Check whether we have past submission of the EDW deal we're looping over
        if deal.edcode.unique() in unique_disc_deals:

            # If so, check for new submissions of the deal
            filtr = (
                self._trackers.DISC_submissions.edcode == deal.edcode.unique()[0]
            )
            if len(self._trackers.DISC_submissions.loc[filtr]) < len(deal):

                # If so, push new submissions through Pipeline 1A
                self.update_deals(EDW_submissions=deal)

        else:
            self.update_deals(EDW_submissions=deal)

    else:
        self.update_deals(EDW_submissions=deal)

    print(">>> All", deal.edcode.iloc[0], "submissions have been downloaded!")

def run_pipeline_3A_4A(self, new_tables=False):
    """Activates pipelines 3A and 4A, running between the DISC HDFS environment and the front-end Impala tables

    Parameters
    ----------
    new_tables : bool, optional
        [Whether to create new tables or to update existing ones], by default False
    """
    # print(f"\n >> Initializing Pipelines 3A & 4A")
    self.title_generator("Initializing Pipelines 3A & 4A")

    # Warn user that this isn't a testrun
    if not self._testrun and new_tables:
        inp = input(
            "Please note that you are NOT running a testrun.\n Would you like to continue? [y/n]"
        ).lower()

    # Proceed with caution
    if self._testrun or inp in ["y", "yes"] or not new_tables:

        # Disable proxies in order to connect with DISC
        if self._connections.proxies.HTTPS != None:
            self.initialize_proxy(delete=True)

        self.create_tables(new_tables=new_tables)

@staticmethod
def serializer(storage, data):
    """Adds data to NameSpace objects.

    Parameters
    ----------
    storage : NameSpace object
        NameSpace object that may contain data already
    data : dictionary
        New data to add to the storage

    Returns
    -------
    NameSpace object
        returns the storage with updated data
    """

    # Step 1: Convert namespace to dict
    storage = json.loads(json.dumps(storage, default=lambda item: vars(item)))
    # Step 2: Add key-value pairs
    storage = {**storage, **data}
    # Step 3: Convert back to SimpleNamespace object
    storage = json.loads(
        json.dumps(storage), object_hook=lambda item: SimpleNamespace(**item)
    )

    return storage

def setup_logger(name, log_file, level=logging.INFO):
    """Creates loggers in an easy and flexible way

    Parameters
    ----------
    name : str
        name of the logger
    log_file : str
        path/filename of the logger
    level : loglevel, optional
        loglevel to describe to the logger, by default logging.INFO

    Returns
    -------
    logger
        Custom made logger
    handler
        Handler of the logger
    """

    formatter = logging.Formatter("%(asctime)s:%(levelname)s:%(message)s")
    handler = logging.FileHandler(log_file)
    handler.setFormatter(formatter)

    logger = logging.getLogger(name)
    logger.setLevel(level)
    logger.addHandler(handler)

    return logger, handler

    df : Pandas dataFrame
        Dataframe to upload to DARWIN
    darwin_file_name : str, optional
        Can be different from the local file name, by default "disc_submissions_rmb.csv"
    darwin_root_folder_id : int, optional
        node of the root folder (last set of digits of the DARWIN folder URL), by default 290298255
    darwin_sub_folder : str, optional
        use None or empty string if uploading directly to the root folder, by default "EDWtoDISC"
    to_csv_args : dict, optional
        a dictionary with all the args to pass to pandas.to_csv, by default {"header": "True", "index": "False"}
    file_exists_policy : str, optional
        When uploading a file - or a set of files - one can choose what to do when a file with the same name already exists in the target Darwin folder. The following options are available:
        NEW_VERSION: Is the default behaviour. Adds a new version to the existing file. The darwin node_id will be the same.
        DELETE_OLD: Deletes the existing darwin file (all versions) and adds the new file as a new darwin node. The darwin node_id will be a new one.
        KEEP_OLD: It doesn't upload the new file. The existing darwin file is kept as it is.,
        by default "NEW_VERSION"
    """
    darwin_disseminator = darwin_dis(darwin_root_folder_id)
    darwin_disseminator.upload_df_csv(
        darwin_sub_folder,
        darwin_file_name,
        df,
        to_csv_args,
        file_exists_policy,
    )
